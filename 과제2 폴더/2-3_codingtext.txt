#include <string>
#include <vector>

using namespace std;

vector<int> solution(vector<int> num_list, int n) {
    vector<int> answer;
    int len = num_list.size();
    answer.resize(len);
    
    int j = 0;
    for(int i = n; i < len; i++, j++) {
        answer[j] = num_list[i];
    }
    
    for(int i = 0; i < n; i++, j++) {
        answer[j] = num_list[i];
    }
    
    return answer;
}
