#include <string>
#include <vector>

using namespace std;

int solution(vector<vector<int>> board, int k) {
    int answer = 0;
    int x = board.size(), y = board[0].size();
    
    for(int i = 0 ; i < x; i++) {
        for(int j = 0 ; j < y; j++) {
            answer += (((i + j) <= k) ? board[i][j] : 0); 
        }
    }
    return answer;
}
