#include <string>
#include <vector>

using namespace std;

int solution(string my_string, string is_prefix) {
    int my_len = my_string.length();
    int is_len = is_prefix.length();

    if (my_len - is_len < 0) {
        return 0;
    }
    
    if(my_string.at(0) != is_prefix.at(0)) {
        return 0;
    }
    
    for(int i = 1; i <= my_len; i++) {
        string sub = my_string.substr(0, i+1);
        
        if(!sub.compare(is_prefix))
            return 1;
    }
    
    return 0;
}
